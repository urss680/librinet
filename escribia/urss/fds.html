        const ASSET_SOURCE_URL = 'https://raw.githubusercontent.com/LIBRINETURL/edunube/refs/heads/main/Librillaia%20images/ACTUAL.html';

        function sanitizeUrl(url) {
            if (!url) return '';
            return url.replace(/^['"]|['"]$/g, '').trim();
        }

        async function loadDynamicAsset() {
            let finalImageUrl = null;
            let finalClickUrl = null;

            try {
                const response = await fetch(ASSET_SOURCE_URL);

                if (!response.ok) {
                    imageElement.src = 'invalid-url';
                    return;
                }

                const assetText = await response.text();
                const trimmedAssetText = assetText.trim();
                const match = trimmedAssetText.match(/^(.+?)\s*\((.+?)\)$/);

                if (match && match.length === 3) {
                    finalImageUrl = match[1].trim();
                    finalClickUrl = sanitizeUrl(match[2]);
                } else {
                    finalImageUrl = trimmedAssetText;
                }
            } catch (error) {
                imageElement.src = 'invalid-url';
                return;
            }

            if (finalImageUrl) {
                imageElement.src = finalImageUrl;
                imageElement.onload = () => {
                    linkElement.style.display = 'block';
                };
            } else {
                imageElement.src = 'invalid-url';
            }

            if (finalClickUrl) {
                linkElement.href = finalClickUrl;
            }
        }

        async function searchWikipedia(query) {
            if (!query.trim()) return;

            searchButton.disabled = true;
            resultsContainer.innerHTML = '<div class="loading">Buscando...</div>';

            try {
                const searchUrl = `https://es.wikipedia.org/w/api.php?action=query&list=search&srsearch=${encodeURIComponent(query)}&format=json&origin=*&srlimit=10`;
                
                const response = await fetch(searchUrl);
                const data = await response.json();
                
                if (data.query.search.length === 0) {
                    resultsContainer.innerHTML = '<div class="no-results">No se encontraron resultados.</div>';
                    return;
                }

                const pageIds = data.query.search.map(result => result.pageid).join('|');
                const extractUrl = `https://es.wikipedia.org/w/api.php?action=query&pageids=${pageIds}&prop=extracts|pageimages|info&exintro&exlimit=max&piprop=original&inprop=url&format=json&origin=*`;
                
                const extractResponse = await fetch(extractUrl);
                const extractData = await extractResponse.json();
                
                displayResults(data.query.search, extractData.query.pages);
            } catch (error) {
                resultsContainer.innerHTML = '<div class="error">Error al buscar. Por favor, intenta de nuevo.</div>';
                console.error('Error:', error);
            } finally {
                searchButton.disabled = false;
            }
        }

        function displayResults(searchResults, extractData) {
            resultsContainer.innerHTML = '<div class="results"></div>';
            const resultsDiv = resultsContainer.querySelector('.results');

            searchResults.forEach(result => {
                const pageId = result.pageid;
                const pageData = extractData[pageId];
                const extract = pageData?.extract || result.snippet + '...';
                
                let imageUrl = null;
                if (pageData?.original?.source) {
                    imageUrl = pageData.original.source;
                } else if (pageData?.thumbnail?.source) {
                    imageUrl = pageData.thumbnail.source;
                }
                
                const resultItem = document.createElement('div');
                resultItem.className = 'result-item';
                resultItem.onclick = () => loadFullArticle(result.title);
                
                resultItem.innerHTML = `
                    <div class="result-image">
                        ${imageUrl ? 
                            `<img src="${imageUrl}" alt="${result.title}" onerror="this.style.display='none'; this.nextElementSibling.style.display='block';">` : 
                            '<div class="no-image">Sin imagen</div>'
                        }
                    </div>
                    <div class="result-content">
                        <div class="result-title">${result.title}</div>
                        <div class="result-snippet">${extract}</div>
                        <div style="color: #8f94fb; font-size: 14px; margin-top: 10px;">Clic para leer el artículo completo →</div>
                    </div>
                `;
                
                resultsDiv.appendChild(resultItem);
            });
        }

        async function loadFullArticle(title) {
            articleContent.innerHTML = '<div class="loading">Cargando artículo...</div>';
            summaryTopContainer.style.display = 'none';
            articleModal.style.display = 'block';
            
            try {
                const url = `https://es.wikipedia.org/w/api.php?action=parse&page=${encodeURIComponent(title)}&format=json&origin=*&prop=text|images&imlimit=max`;
                
                const response = await fetch(url);
                const data = await response.json();
                
                if (data.parse && data.parse.text) {
                    let content = data.parse.text['*'];
                    content = cleanArticleContent(content);
                    currentArticleText = extractTextFromHTML(content);
                    
                    if (data.parse.images && data.parse.images.length > 0) {
                        await processArticleImages(data.parse.images, title);
                    }
                    
                    makeLinksClickable();
                    
                    // Mostrar el artículo primero
                    articleContent.innerHTML = content;
                    
                    // Luego generar el resumen después de mostrar el artículo
                    await generateSummary();
                } else {
                    articleContent.innerHTML = '<div class="error">No se pudo cargar el artículo.</div>';
                }
            } catch (error) {
                articleContent.innerHTML = '<div class="error">Error al cargar el artículo.</div>';
                console.error('Error:', error);
            }
        }

        async function generateSummary() {
            if (!currentArticleText) return;
            
            try {
                const chunks = [];
                const chunkSize = 5000;
                for (let i = 0; i < currentArticleText.length; i += chunkSize) {
                    chunks.push(currentArticleText.substring(i, i + chunkSize));
                }
                
                let summaries = [];
                
                for (let i = 0; i < chunks.length; i++) {
                    const apiUrl = `https://text.pollinations.ai/${encodeURIComponent(chunks[i])}?system=Genera un resumen conciso de esta parte del artículo`;
                    
                    const response = await fetch(apiUrl);
                    const summary = await response.text();
                    
                    if (summary && summary.trim()) {
                        summaries.push(`Parte ${i + 1}: ${summary}`);
                    }
                }
                
                if (summaries.length > 0) {
                    let finalSummary;
                    
                    if (summaries.length === 1) {
                        finalSummary = summaries[0];
                    } else {
                        const combinedSummaries = summaries.join('\n\n');
                        const finalApiUrl = `https://text.pollinations.ai/${encodeURIComponent(combinedSummaries)}?system=Combina estos resúmenes en un resumen final coherente del artículo completo`;
                        
                        const finalResponse = await fetch(finalApiUrl);
                        finalSummary = await finalResponse.text();
                        
                        if (!finalSummary || !finalSummary.trim()) {
                            finalSummary = summaries.join('\n\n');
                        }
                    }
                    
                    summaryTopContainer.innerHTML = `
                        <h3>📋 Resumen del artículo:</h3>
                        <p>${finalSummary}</p>
                    `;
                    summaryTopContainer.style.display = 'block';
                }
            } catch (error) {
                console.error('Error al generar el resumen:', error);
            }
        }

        function extractTextFromHTML(html) {
            const temp = document.createElement('div');
            temp.innerHTML = html;
            return temp.textContent || temp.innerText || '';
        }

        function cleanArticleContent(content) {
            const temp = document.createElement('div');
            temp.innerHTML = content;
            
            const unwanted = temp.querySelectorAll('.infobox, .navbox, .metadata, .dablink, .rellink, .catlinks, .magnify, .NavFrame, .noprint, .ambox, .reflist, .references');
            unwanted.forEach(el => el.remove());
            
            const allElements = temp.querySelectorAll('*');
            allElements.forEach(el => {
                el.removeAttribute('style');
                if (el.tagName !== 'IMG' && el.tagName !== 'A') {
                    el.removeAttribute('class');
                }
            });
            
            const images = temp.querySelectorAll('img');
            images.forEach(img => {
                img.style.maxWidth = '100%';
                img.style.height = 'auto';
            });
            
            return temp.innerHTML;
        }

        function makeLinksClickable() {
            const links = articleContent.querySelectorAll('a');
            links.forEach(link => {
                const href = link.getAttribute('href');
                if (href && href.startsWith('/wiki/')) {
                    link.onclick = (e) => {
                        e.preventDefault();
                        const pageName = decodeURIComponent(href.replace('/wiki/', ''));
                        loadFullArticle(pageName);
                    };
                } else if (href && href.startsWith('#')) {
                    link.onclick = (e) => {
                        e.preventDefault();
                        const target = document.querySelector(href);
                        if (target) {
                            target.scrollIntoView({ behavior: 'smooth' });
                        }
                    };
                } else {
                    link.target = '_blank';
                    link.rel = 'noopener noreferrer';
                }
            });
        }

        async function processArticleImages(images, articleTitle) {
            const imageUrls = await getImageUrls(images, articleTitle);
            const imgs = articleContent.querySelectorAll('img');
            imgs.forEach((img, index) => {
                if (imageUrls[index]) {
                    img.src = imageUrls[index];
                    img.style.display = 'block';
                    img.style.maxWidth = '100%';
                    img.style.height = 'auto';
                    img.style.margin = '15px 0';
                    img.style.borderRadius = '8px';
                    img.style.boxShadow = '0 5px 15px rgba(0,0,0,0.3)';
                }
            });
        }

        async function getImageUrls(images, articleTitle) {
            const imageUrls = [];
            
            for (const imageName of images.slice(0, 10)) {
                try {
                    const url = `https://es.wikipedia.org/w/api.php?action=query&prop=imageinfo&iiprop=url&titles=${encodeURIComponent('File:' + imageName)}&format=json&origin=*`;
                    const response = await fetch(url);
                    const data = await response.json();
                    
                    const pages = data.query.pages;
                    const pageId = Object.keys(pages)[0];
                    if (pages[pageId].imageinfo && pages[pageId].imageinfo[0].url) {
                        imageUrls.push(pages[pageId].imageinfo[0].url);
                    }
                } catch (error) {
                    console.error('Error al obtener URL de imagen:', error);
                }
            }
            
            return imageUrls;
        }
